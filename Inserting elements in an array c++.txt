
#include <iostream>
#include <array>

using namespace std;

int main()
{
   int arr[]={1,5,6,7,4,9,3,2};
   
  int n;
  cin>>n;
   
  int number;
  cin>>number;
   
  for(int i=sizeof(arr)/sizeof(arr[0])-1;i>=n-1;i--){
      
      arr[i+1]=arr[i];
      
  }
  
    arr[n-1]=number;
  

  for(int i=0;i<=sizeof(arr)/sizeof(arr[0]);i++){
      
      cout<<arr[i]<<" ";
  }




    return 0;
}
